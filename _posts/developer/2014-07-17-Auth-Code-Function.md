```php

<?php

/**
 * class
 */
class Foo
{
    public $a;
    
    private $c = null;
    
    public function __construct()
    {
        $this->c = "test";
    }
    
    public function bar()
    {
        $foobar = "this is a test for highlight";
        
        return $foobar;
    }
    
    public function get()
    {
        return $this->c . ':' . $this->bar();
    }
}

$f = new Foo;
$f->get();
?>
```
